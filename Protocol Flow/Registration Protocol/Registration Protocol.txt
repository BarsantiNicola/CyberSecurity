[websequencediagram]
title Registration Protocol

client->server: SIGN_UP
server->client: SIGN_UP_OK / SIGN_UP_FAIL

[message format]

SIGN_UP
{
	Message_Type: 1 byte
	Client_Certificate: 0-4096 byte
	Timestamp: 8 byte
}CIFRED[Server RSA public key]


SIGN_OK
{
	Message_type: 1 byte
	Timestamp: 8 byte
}CIFRED[Server RSA private key]


SIGN_FAIL
{
	Message_type: 1 byte
	Timestamp: 8 byte
}CIFRED[Server RSA private key]
